import matplotlib.pyplot as plt

for col in df.columns:
    if col not in ['Bullied_in_school', 'Bullied_outside_school',
                   'Cyber_bullied']:
        values = df.groupby(['Bullied_in_school', 'Bullied_outside_school',
                             'Cyber_bullied'])[col].get_group(('Yes', 'Yes', 'Yes')).value_counts(normalize=True) * 100

        plt.title(f'Percentage of {col} for (Yes, Yes, Yes)')
        plt.xlabel(col)
        plt.ylabel('Percentage')
        plt.show()
